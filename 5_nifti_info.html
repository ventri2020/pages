NIfTI B&W and DICOM COLOR IMAGES (PoC)

See What is a non-capturing group in regular expressions?.

kind = "image_base|nifti_color|dicom_color"

splitter = glue::glue(
  "^(5_nifti)/([0-9]+)/({kind})(:?/.*/.*)?/.*\\.(?:img|dcm)$"
)
column_names = c("series", "inumber", "kind", "tmp")
base_images = list.files(
  path = "5_nifti",
  pattern = ".*\\.(img|dcm)$",
  full.names = TRUE,
  recursive = TRUE
) %>%
  stringr::str_subset(kind)

nifti_info(base_images, into = column_names, regex = splitter) %>%
  dplyr::select(
    series, inumber, kind, `xyz`, `pixel resolution [mm]`
  ) %>%
  dplyr::arrange(inumber, kind)
